<script>
    // Javascript остался прежним, так как логика полностью соответствует новым требованиям
    let currentStep = 0;
    const totalSteps = 7;

    const praises = [
        "Здорово получилось!",
        "Ты заметил важную деталь.",
        "Отличный ход мыслей!",
        "Твоя логика работает!",
        "Очень внимательно!"
    ];

    function updateProgress() {
        const percent = (currentStep / totalSteps) * 100;
        document.getElementById('progress-fill').style.width = percent + '%';
    }

    function showHint(step, text) {
        const hintDiv = document.getElementById('hint-text');
        hintDiv.style.display = 'inline-block';
        hintDiv.innerText = text;
        // Подсказка висит чуть дольше для спокойного чтения
        setTimeout(() => {
            hintDiv.style.display = 'none';
            hintDiv.innerText = '';
        }, 5000);
    }

    function nextStep(step) {
        document.querySelectorAll('.screen').forEach(s => s.classList.add('hidden'));
        document.getElementById(`screen-${step}`).classList.remove('hidden');
        currentStep = step;
        updateProgress();
        document.getElementById('feedback').innerText = '';
        document.getElementById('hint-text').style.display = 'none';
    }

    function checkAnswer(step, answer) {
        const feedback = document.getElementById('feedback');
        let correct = false;

        if (step === 1 && answer === 2) correct = true;
        if (step === 2 && answer === 'Миша') correct = true;
        if (step === 3 && answer === '7 - 4 + 2 + 5') correct = true;
        if (step === 4 && answer === 3) correct = true;
        if (step === 5 && answer === 8) correct = true;
        if (step === 6 && answer === 'correct') correct = true;
        if (step === 7) {
            const story = document.getElementById('final-story').value;
            if (story.length > 3) {
                document.getElementById('user-conclusion').innerText = "Твоя версия: " + story;
                correct = true;
            } else {
                showHint(7, "Детектив думает: давай запишем хотя бы пару слов для истории...");
                return;
            }
        }

        if (correct) {
            const randomPraise = praises[Math.floor(Math.random() * praises.length)];
            feedback.innerText = randomPraise;
            // Пауза перед переходом чуть длиннее для спокойствия
            setTimeout(() => {
                nextStep(step + 1);
            }, 18
